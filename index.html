local Players = game:GetService("Players") local player = Players.LocalPlayer local timerLabel = Instance.new("TextLabel") timerLabel.Size = UDim2.new(0, 200, 0, 50) timerLabel.Position = UDim2.new(0.5, -100, 0.5, -25) timerLabel.TextScaled = true timerLabel.Parent = player:WaitForChild("PlayerGui") local countdownTime = 10 * 24 * 60 * 60 -- 10 days in seconds local startTime = os.time() while countdownTime > 0 do local elapsedTime = os.time() - startTime local remainingTime = countdownTime - elapsedTime local days = math.floor(remainingTime / (24 * 60 * 60)) local hours = math.floor((remainingTime % (24 * 60 * 60)) / (60 * 60)) local minutes = math.floor((remainingTime % (60 * 60)) / 60) local seconds = remainingTime % 60 timerLabel.Text = string.format("%dd %dh %dm %ds", days, hours, minutes, seconds) wait(1) end timerLabel.Text = "EXPIRED"
